finite_differences <- function(x, y) {
  n <- length(x)
  derivativeFunction <- vector(length = n)
  for (i in 2:n) {
    derivativeFunction[i-1] <- (y[i-1] - y[i]) / (x[i-1] - x[i])
  }
  derivativeFunction[n] <- (y[n] - y[n - 1]) / (x[n] - x[n - 1])
  return(derivativeFunction)
}

We generate some data from the following model:

\[ X_i \mid \lambda_i \sim \text{Weibull}(\alpha, \lambda_i) \] where \(\lambda_i\) is defined as:

\[ \lambda_i = \xi_i \gamma \alpha e^{\beta^T z_i} t^{\alpha - 1} \] and \(\xi_i\) is distributed \(\text{Gamma}(\nu^{-1}, \nu^{-1})\) so that \(\Var{\xi_i} = \nu\) and \(\Exp{\xi_i} = 1\).

set.seed(123)
p <- 2
n <- 1e3
X <- matrix(rnorm(p * n),n,p)
re <- rgamma(n,1/5,1/5)
beta <- c(0.25, -0.25)
gamma <- 1/5
alpha <- 2
haz <- re * exp(X %*% beta)
x <- sapply(haz, \(x) 
            rweibull(1, 
                     shape = alpha, 
                     scale = (1/(x * gamma))^(1/alpha))
            )
cens <- rexp(n,1/5)
t <- pmin(x, cens)
status <- as.integer(x <= cens)
sdat <- 
  list(
    N = n,
    p = p,
    X = X,
    t = t,
    delta = status
  )

df_sim <- data.frame(
  X_1 = X[,1],
  X_2 = X[,2],
  time = t,
  status = status,
  id = 1:nrow(X)
)

We fit a Cox model to the data:

ph <- coxph(Surv(time, status) ~ X_1 + X_2, df_sim)
df_new <- data.frame(X_1 = 0, X_2 = 0)
hz <- basehaz(ph, df_new)

deri <- finite_differences(x = hz$time, y = hz$hazard)
hz$baseline <- deri

# Non-parametric smooth curve for estimating baseline hazard function over time
smooth_baseline <- ss(x = hz$time, y = hz$baseline, df = 4)
true_baseline <- alpha * gamma * smooth_baseline$x ^ (alpha - 1)
ylims <- range(true_baseline)

with(smooth_baseline, 
     plot(x,y, 
          type = "l", 
          ylim = ylims, col = "blue", 
          main = "True baseline hazard (red) vs. inferred hazard (blue)", 
          ylab = bquote(lambda[t]), 
          xlab = "time")
     )
with(smooth_baseline, lines(x, true_baseline, col = "red"))
